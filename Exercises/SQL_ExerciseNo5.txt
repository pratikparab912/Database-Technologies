show databases;
use exercise;

select Sname,QTY from SUPPLIER,SUPPLIER_PARTS_PROJECTS
where SUPPLIER.Sno=SUPPLIER_PARTS_PROJECTS.Sno;

select Pname,QTY from PARTS,SUPPLIER_PARTS_PROJECTS
where PARTS.Pno=SUPPLIER_PARTS_PROJECTS.Pno;

select Jname,QTY from PROJECTS,SUPPLIER_PARTS_PROJECTS
where PROJECTS.Jno=SUPPLIER_PARTS_PROJECTS.Jno;

select Sname,Pname,Jname,QTY
from SUPPLIER,PARTS,PROJECTS,SUPPLIER_PARTS_PROJECTS
where SUPPLIER.Sno=SUPPLIER_PARTS_PROJECTS.Sno
AND PARTS.Pno=SUPPLIER_PARTS_PROJECTS.Pno
AND PROJECTS.Jno=SUPPLIER_PARTS_PROJECTS.Jno;

select Sname,Pname,Jname 
from SUPPLIER,PARTS,PROJECTS
where SUPPLIER.City=PROJECTS.City
group by Sname;

select Pname,QTY
from PARTS,SUPPLIER_PARTS_PROJECTS
where Color='red'
and PARTS.Pno=SUPPLIER_PARTS_PROJECTS.Pno;

select Sname, QTY
from SUPPLIER,SUPPLIER_PARTS_PROJECTS
where Status=20
and SUPPLIER.Sno=SUPPLIER_PARTS_PROJECTS.Sno;

select Pname,QTY
from PARTS,SUPPLIER_PARTS_PROJECTS
where Weight > 14
and PARTS.Pno=SUPPLIER_PARTS_PROJECTS.Pno;

select Jname,City
from PROJECTS,SUPPLIER_PARTS_PROJECTS
where QTY > 500
and PROJECTS.Jno=SUPPLIER_PARTS_PROJECTS.Jno;

select Pname,QTY
from PARTS,SUPPLIER_PARTS_PROJECTS
where Weight < 15
and PARTS.Pno=SUPPLIER_PARTS_PROJECTS.Pno;
